    \section{Particle-Mesh Coupling}

    \subsection{Smoothing particle splines}

    The Particle-Mesh coupling algorithms are based on a spline smoothing of the particle Dirac distribution in the discrete Lagrangian functional.
    Associated with a particle with arbitrary position $\mathbf{x}_p = (x_p, y_p, z_p)$, the \textbf{particle spline} is a tensor product 
    $$
    S_h( \mathbf{x}-\mathbf{x}_p) = S_{\Delta x}^{d_x-1}(x-x_p)S_{\Delta y}^{d_y-1}(y-y_p)S_{\Delta z}^{d_z-1}(z-z_p)
    $$
    involving univariate scaled splines of the form
    $S_h^d(x)=\frac 1h S^d(\frac {x}{h})$,
    where $S^d$ is the centered, reference cardinal B-spline of degree $d$ defined recursively by 
    $$S^0(x)= \begin{cases} 1  &\text{ if } ~ -\frac{1}{2} \leq x \leq \frac{1}{2} \\ 0 & \text{ else} \end{cases}$$
    $$ S^d(x) = S^0 * S^{d-1} (x) = \int_{-\frac{1}{2}}^{\frac{1}{2}} S^{d-1} (x-x') \dd x'.$$
    \textbf{Properties of the cardinal B-spline}
    \begin{itemize}
        \item The cardinal splines are even $S^d(x) = S^d(-x)$
        \item The support of $S^d$ is $\left[-\frac{d+1}{2},\frac{d+1}{2} \right]$
        \item Both the reference and scaled cardinal splines have unit integral:
        $$ \int_{-\infty}^{\infty} S^d(x) \dd x = \int_{-\infty}^{\infty} S_h^d(x) \dd x = 1$$
        \item Denoting grid nodes by $x_i = ih$, the scaled cardinal B-spline verifies
        \begin{equation} \label{splinerel}
            \int_{x_{i-\frac{1}{2}}}^{x_{i+\frac{1}{2}}} S_h^{d-1}(x-x_p)\dd x = hS_h^d(x_i-x_p) = S^d\Big(\frac {x_i-x_p}{h}\Big)
        \end{equation}
    \end{itemize}

    \subsection{Node and cell splines at particle positions}

    The coupling scheme then involves two kinds of univariate B-splines, evaluated at particle positions:
    \begin{itemize}
        
        \item \textbf{node splines} centered on grid nodes, of the form
        $$ 
        S^\node_i(x_p) := S^d \Big(\frac{x_p-x_i}{h}\Big)
        $$
        This spline is supported on the interval $\left[x_i -\frac{(d+1)h}{2}, x_i + \frac{(d+1)h}{2} \right]$,
        and its values are independent of the grid size.

        \item \textbf{cell splines} centered on cell midpoints, of the form
        $$ 
        S^\cell_i(x_p) := \frac 1h S^{d-1}\Big(\frac{x_p - x_\iph}{h}\Big)
        $$
        This spline is centered on the supported on the interval $\left[x_\iph -\frac{dh}{2}, x_\iph + \frac{dh}{2} \right]$,
        and its cell-integrals are independent of the grid size.

    \end{itemize}
    These splines naturally appear in the variational scheme since the above relation \eqref{splinerel} implies that 
    the dual-node (resp. dual-cell) degree of freedom of a univariate particle spline is a cell spline (resp. node spline) evaluated
    at the particle position.


    \subsection{Reduction of particle splines}

    The reduction operators on the dual mesh for our smoothing function constructed from the cardinal B-Spline can be computed based on the properties above. 
    Specifically, dual geometric degrees of freedom are product of univariate cell and node splines evaluated at the particles:
    $$
    \begin{aligned}
        \tcR_0(S_h(\mathbf{x}-\mathbf{x}_p))_{\iph,\jph,\kph} &= S^\cell_i(x_p)\,S^\cell_j(y_p) \, S^\cell_k(z_p)\\
        \tcR_1(\mathbf{e}_x S_h(\mathbf{x}-\mathbf{x}_p))_{i,\jph,\kph} &= S^\node_i(x_p)\,S^\cell_j(y_p) \, S^\cell_k(z_p)\\
        \tcR_1(\mathbf{e}_y S_h(\mathbf{x}-\mathbf{x}_p))_{\iph,j,\kph} &= S^\cell_i(x_p)\,S^\node_j(y_p) \, S^\cell_k(z_p)\\
        \tcR_1(\mathbf{e}_z S_h(\mathbf{x}-\mathbf{x}_p))_{\iph,\jph,k} &= S^\cell_i(x_p)\,S^\cell_j(y_p) \, S^\node_k(z_p)\\
        \tcR_2(\mathbf{e}_x S_h(\mathbf{x}-\mathbf{x}_p))_{\iph,j,k} &= S^\cell_i(x_p)\,S^\node_j(y_p) \, S^\node_k(z_p)\\
        \tcR_2(\mathbf{e}_y S_h(\mathbf{x}-\mathbf{x}_p))_{i,\jph,k} &= S^\node_i(x_p)\,S^\cell_j(y_p) \, S^\node_k(z_p)\\
        \tcR_2(\mathbf{e}_z S_h(\mathbf{x}-\mathbf{x}_p))_{i,j,\kph} &= S^\node_i(x_p)\,S^\node_j(y_p) \, S^\cell_k(z_p)\\
        \tcR_3(S_h(\mathbf{x}-\mathbf{x}_p))_{i,j,k}                &= S^\node_i(x_p)\,S^\node_j(y_p) \, S^\node_k(z_p).
    \end{aligned}
    $$

    \subsection{Smoothed particle quantities}

    The smoothed particle distribution function is of the form
    $$ 
    f(t,\mathbf{x},\mathbf{v}) = \sum_p w_p S_h(\mathbf{x}-\mathbf{x}_p(t))\delta (\mathbf{v}-\mathbf{v}_p(t)).
    $$
    It follows that the particle charge and current densities are respectively
    $$
    \rho(t,\mathbf{x}) = \sum_p q_p w_p S_h(\mathbf{x}-\mathbf{x}_p(t)), ~~~~ \bJ(t,\mathbf{x}) = \sum_p q_p w_p \mathbf{v}_p(t) S_h(\mathbf{x}-\mathbf{x}_p(t)).
    $$
    In the GEMPIC theory, these two densities play the role of Maxwell source terms through their dual degrees of freedom 
    on the grid.
    
    Specifically, the particle charge density $\rho$ is represented as a 3-form on the dual mesh which is 
    characterized by the dual cell integrals $\tilde{\arr{\rho}} = \tcR_3(\rho)$ in $\tilde{ \cC}_3$,
    By linearity of the reduction operators, these degrees of freedom %(the dual cell integrals) 
    read %of $\rho$, which are its coefficients in $\tilde{ \cC}_3$, are 
    $$
    \tilde{\arr{\rho}}_{i,j,k}(t) := \tcR_3(\rho(t))_{i,j,k}
    = \sum_p q_p w_p S^\node_i(x_p(t))\, S^\node_j(y_p(t)) \,S^\node_k(z_p(t)).
    $$
    As for the particle current density $\bJ$, it is represented as a 2-form on the dual mesh. 
    Thus it is characterized 
    by its fluxes through all the faces of the mesh, gathered in the vector 
    $\tilde{\arrJ} = \tcR_2( \mathbf{J})$ in $\tilde{ \cC}_2$. These coefficients read
        % imilarly the coefficients of $\mathbf{J}$ in $\tilde{ \cC}_2$ are 
    $$
    \begin{aligned}
    \tilde{\arrJ}_{\iph,j,k}(t) &:= \tcR_2( \mathbf{J}(t))_{\iph,j,k} = \sum_p q_p w_p v_{p,x}(t) S^\cell_i(x_p(t))\, S^\node_j(y_p(t)) \,S^\node_k(z_p(t)) \\ 
    \tilde{\arrJ}_{i,\jph,k}(t) &:= \tcR_2( \mathbf{J}(t))_{i,\jph,k} =\sum_p q_p w_p v_{p,y}(t) S^\node_i(x_p(t))\, S^\cell_j(y_p(t)) \,S^\node_k(z_p(t)) \\
    \tilde{\arrJ}_{i,j,\kph}(t) &:= \tcR_2( \mathbf{J}(t))_{i,j,\kph} = \sum_p q_p w_p v_{p,z}(t) S^\node_i(x_p(t))\, S^\node_j(y_p(t)) \,S^\cell_k(z_p(t)) \\
    \end{aligned}
    $$
    which correspond respectively to the three components of $\mathbf{J}$, i.e.  $J_x$, $J_y$, $J_z$ 

    \subsection{Smoothed electromagnetic fields}

The discrete electric field $\bE$ is characterized by its integrals on the edges of the primal grid, % denoted by $\arrE$, where 
\begin{itemize}
    \item $\arrE_{\iph,j,k} \approx \int_{x_i}^{x_{i+1}} E_x(x,y_j,z_k)\dd x $
    \item $\arrE_{i,\jph,k} \approx \int_{y_j}^{y_{j+1}} E_y(x_i,y,z_k)\dd y $
    \item $\arrE_{\iph,j,k} \approx \int_{z_k}^{z_{k+1}} E_z(x_i,y_j,z)\dd z $
\end{itemize}
One could define (for instance using a finite element setting) an electric field that is defined everywhere and 
has exactly these geometric degrees of freedom. 
However this is not necessary: in the GEMPIC scheme indeed the particles are coupled with a 
spline-smoothed electric field $\bE^S$ which only involves the above primal degrees of freedom $\arrE \in \cC_1$
and the reduction $\tcR_2$ of the (vectorized) particle splines, namely:
$$
\mathbf{e}_{d} \cdot \bE^S(\mathbf{x}_p) 
:=  \arrE \cdot \tcR_2\big(\mathbf{e}_{d} \, S_h(\mathbf{x}-\mathbf{x}_p)\big)
= \sum_{\texttt{e} \in \{\text{edges}\}} \arrE_{\texttt{e}} \tcR_2\big(\mathbf{e}_{d} \, S_h(\mathbf{x}-\mathbf{x}_p)\big)_{\tilde{\texttt{f}}(\texttt{e})}
$$
for all dimension $d \in \{x,y,z\}$, 
where $\mathbf{e}_{d}$ is the canonical basis vector of $\mathbb{R}^3$ along dimension $d$.
and for each primal edge $\texttt{e}$ we have denoted by $\tilde{\texttt{f}}(\texttt{e})$ its dual face.
Given the reduction formulas above, this yields for the different components
$$
\begin{aligned}
    E^S_x(\mathbf{x}_p) &= 
            \sum_{i,j,k} \arrE_{\iph,j,k} S^\cell_i(x_p)\,S^\node_j(y_p) \, S^\node_k(z_p) \\
    E^S_y(\mathbf{x}_p) &= 
            \sum_{i,j,k} \arrE_{i,\jph,k} S^\node_i(x_p)\,S^\cell_j(y_p) \, S^\node_k(z_p) \\
    E^S_z(\mathbf{x}_p) &= 
            \sum_{i,j,k} \arrE_{i,j,\kph} S^\node_i(x_p)\,S^\node_j(y_p) \, S^\cell_k(z_p)\\
\end{aligned}
$$

In the same way the discrete magnetic field $\bB$ is characterized by its fluxes through the faces of the primal grid, 
denoted 
\begin{itemize}
    \item $\arrB_{\iph,j,k} \approx \int_{y_j}^{y_{j+1}}\int_{z_k}^{z_{k+1}} B_x(x_i,y,z)\dd y \dd z $
    \item $\arrB_{i,\jph,k} \approx \int_{x_i}^{x_{i+1}}\int_{z_k}^{z_{k+1}} B_y(x,y_j,z)\dd x \dd z $
    \item $\arrB_{\iph,j,k} \approx \int_{x_i}^{x_{i+1}}\int_{y_j}^{y_{j+1}} B_z(x,y,z_k)\dd x\dd y $
\end{itemize}
In the variational GEMPIC scheme the particles are coupled with a spline-smoothed magnetic field 
which involves the primal degrees of freedom $\arrB \in \cC_2$
and the reduction $\tcR_1$ of the (vectorized) particle splines, namely:
$$
\mathbf{e}_{d} \cdot \bB^S(\mathbf{x}_p) 
:=  \arrB \cdot \tcR_1\big(\mathbf{e}_{d} \, S_h(\mathbf{x}-\mathbf{x}_p)\big)
= \sum_{\texttt{f} \in \{\text{faces}\}} \arrB_{\texttt{f}} \tcR_1\big(\mathbf{e}_{d} \, S_h(\mathbf{x}-\mathbf{x}_p)\big)_{\tilde{\texttt{e}}(\texttt{f})}
$$
again for all dimension $d \in \{x,y,z\}$, 
and where for each primal face $\texttt{f}$ we have now denoted by $\tilde{\texttt{e}}(\texttt{f})$ its dual edge.

Using the formulas above for the dual dofs of the spline particles, this yields for the different components
$$
\begin{aligned}
    B^S_x(\mathbf{x}_p) &= \sum_{i,j,k} \arrB_{i,\jph,\kph} \,S^\node_i(x_p)\,S^\cell_j(y_p) \, S^\cell_k(z_p) \\
    B^S_y(\mathbf{x}_p) &= \sum_{i,j,k} \arrB_{\iph,j,\kph} \,S^\cell_i(x_p)\,S^\node_j(y_p) \, S^\cell_k(z_p) \\
    B^S_z(\mathbf{x}_p) &= \sum_{i,j,k} \arrB_{\iph,\jph,k} \,S^\cell_i(x_p)\,S^\cell_j(y_p) \, S^\node_k(z_p). \\
\end{aligned}
$$

\subsection{Exact computation of time integrated $J_x$, $v_{p,x}B_y(x_p)$ and $v_{p,x}B_z(x_p)$ }

One step of the hamiltonian splitting algorithm, consists in pushing the particles successively in the 3 axis dimensions and updating positions, velocities and electric field leaving the magnetic field constant.
Let us describe this algorithm when the particles are pushed in the $x$-direction. The two other directions are handled similarly.  

In this step the following system needs to be solved
$$
\begin{aligned}
    \fract{x_p}{t} &= v_{p,x} \\
    \fract{v_{y,p}}{t} &= - v_{p,x}(t)B^S_z(x_p(t) ,y_p,z_p) \\
    \fract{v_{z,p}}{t} &= + v_{p,x}B^S_y(x_p(t),y_p,z_p) \\
    \fract{\tilde{\arrD}_{\iph,j,k}}{t} &= -\tilde{\arrJ}_{\iph,j,k}. %\\ 
\end{aligned}
$$
that is,
$$
\begin{aligned}
    \fract{x_p}{t} &= v_{p,x} \\
    \fract{v_{y,p}}{t} &= -v_{p,x} \sum_{i,j,k} \arrB_{\iph,\jph,k} \,S^\cell_i(x_p(t))\,S^\cell_j(y_p) \, S^\node_k(z_p)\\
    \fract{v_{z,p}}{t} &= +v_{p,x} \sum_{i,j,k} \arrB_{\iph,j,\kph} \,S^\cell_i(x_p(t))\,S^\node_j(y_p) \, S^\cell_k(z_p) \\
    \fract{\tilde{\arrD}_{\iph,j,k}}{t} &= - \sum_p q_p w_p v_{p,x} S^\cell_i(x_p(t))\, S^\node_j(y_p) \,S^\node_k(z_p)    
\end{aligned}
$$
while the other variables, not appearing on the left-hand side, stay constant.
Using the first equation $v_{p,x}=\fract{x_p}{t}$, we can write the others as exact time derivatives 
by introducing the primitive of the univariate spline $S^{d_x-1}$ on the reference grid,
that we denote by $\check{S}^{d_x-1}$.
The integration constant is irrelevant as only differences of these primitives will be needed.
Writing
$$
S^\prim_i(x) := \check{S}^{d_x-1}\left(\frac{x-x_\iph}{\Delta x}\right)
$$
gives indeed
$$
\fract{}{t} \Big(S^\prim_i(x_p(t))\Big)
= \frac{v_{p,x}}{\Delta x} S^{d_x-1}\left(\frac{x_p(t)-x_\iph}{\Delta x}\right)
= v_{p,x} S^\cell_i(x_p(t))
$$ 
so that the above system becomes
$$
\begin{aligned}
    \fract{x_p}{t} &= v_{p,x} \\
    \fract{v_{y,p}}{t} &= -\fract{}{t} \left(\sum_{i,j,k} \arrB_{\iph,\jph,k} \,S^\prim_i(x_p(t))\,S^\cell_j(y_p) \, S^\node_k(z_p)\right)\\
    \fract{v_{z,p}}{t} &= +\fract{}{t} \left(\sum_{i,j,k} \arrB_{\iph,j,\kph} \,S^\prim_i(x_p(t))\,S^\node_j(y_p) \, S^\cell_k(z_p) \right)\\
    \fract{\tilde{\arrD}_{\iph,j,k}}{t} &= - \fract{}{t}\left(\sum_p q_p w_p S^\prim_i(x_p(t))\, S^\node_j(y_p) \,S^\node_k(z_p) \right)   
\end{aligned}
$$
Then, integrating the original system in time between $t_0$ and $t_1$ yields without further approximation

$$
\begin{aligned}
    x_p&(t_1) = x_p(t_0) + (t_1-t_0) v_{p,x} \\
    v_{y,p}&(t_1) = v_{y,p}(t_0) -  
    \sum_{i,j,k} \arrB_{\iph,\jph,k} \Big(S^\prim_i(x_p(t_1))-S^\prim_i(x_p(t_0)) \Big) S^\cell_j(y_p) \, S^\node_k(z_p)    
    \\
    v_{z,p}&(t_1) =  v_{z,p}(t_0) + 
    \sum_{i,j,k} \arrB_{\iph,j,\kph} \Big(S^\prim_i(x_p(t_1))-S^\prim_i(x_p(t_0)) \Big) S^\node_j(y_p) \, S^\cell_k(z_p)
    \\
    \tilde{\arrD}_{\iph,j,k} & (t_1) =  \tilde{\arrD}_{\iph,j,k}(t_0) - 
    \sum_p q_p w_p \Big(S^\prim_i(x_p(t_1))-S^\prim_i(x_p(t_0)) \Big) S^\node_j(y_p) \,S^\node_k(z_p).
\end{aligned}
$$
