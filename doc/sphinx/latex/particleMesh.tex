    \section{Particle-Mesh Coupling}

    \subsection{Smoothing function}

    The Particle-Mesh coupling algorithms are based on a smoothing function which is taken to be a tensor product spline
    $S_h( \mathbf{x}) = S_{\Delta x}^{d_x-1}(x)S_{\Delta y}^{d_y-1}(y)S_{\Delta z}^{d_z-1}(z)$, where $S_h^d(x)=\frac 1h S^d(\frac {x}{h})$,
    $S^d$ being the cardinal B-spline defined recursively by 
    $$S^0(x)= 1 \mbox{ if } -\frac{1}{2} \leq x \leq \frac{1}{2}, ~~~ 0 \mbox{ else}.$$
    $$ S^d(x) = S^0 * S^{d-1} (x) = \int_{-\frac{1}{2}}^{\frac{1}{2}} S^{d-1} (x-y) \dd y.$$
    \textbf{Properties of the cardinal B-spline}
    \begin{itemize}
        \item The cardinal splines are even $S^d(x) = S^d(-x)$
        \item The support of $S^d$ is $\left[-\frac{d-1}{2},\frac{d-1}{2} \right]$
        \item The scaled cardinal B-Spline verifies
        $$\int_{x_{i-\frac{1}{2}}}^{x_{i+\frac{1}{2}}} S_h^{d-1}(x-x_p)\dd x = hS_h^d(x_i-x_p),$$
    \end{itemize}

    \subsection{Reduction of the smoothing function at a particle position}

    The reduction operators on the dual mesh for our smoothing function constructed from the cardinal B-Spline can be computed based on the properties above. This yields
    $$
    \begin{aligned}
        \tcR_0(S_h(\mathbf{x}-\mathbf{x}_p))_{\iph,\jph,\kph} &= \frac{1}{\Delta x \Delta y \Delta z}
        S^{d_x-1}\left(\frac{x_\iph-x_p}{\Delta x}\right)\,S^{d_y-1}\left(\frac{y_\jph-y_p}{\Delta y}\right)\,S^{d_z-1}\left(\frac{z_\kph-z_p}{\Delta z}\right) \\
        \tcR_1(S_h(\mathbf{x}-\mathbf{x}_p))_{i,\jph,\kph} &= \frac{1}{\Delta y \Delta z} \,S^{d_x}\left(\frac{x_i-x_p}{\Delta x}\right)\,S^{d_y-1}\left(\frac{y_\jph-y_p}{\Delta y}\right)\,S^{d_z-1}\left(\frac{z_\kph-z_p}{\Delta z}\right) \\
        \tcR_1(S_h(\mathbf{x}-\mathbf{x}_p))_{\iph,j,\kph} &= \frac{1}{\Delta x \Delta z} \, S^{d_x-1}\left(\frac{x_\iph-x_p}{\Delta x}\right)\,S^{d_y}\left(\frac{y_j-y_p}{\Delta y}\right)\,S^{d_z-1}\left(\frac{z_\kph-z_p}{\Delta z}\right) \\
        \tcR_1(S_h(\mathbf{x}-\mathbf{x}_p))_{\iph,\jph,k} &= \frac{1}{\Delta x \Delta y} \,S^{d_x-1}\left(\frac{x_\iph-x_p}{\Delta x}\right)\,S^{d_y-1}\left(\frac{y_\jph-y_p}{\Delta y}\right)\,S^{d_z}\left(\frac{z_k-z_p}{\Delta z}\right) \\
        \tcR_2(S_h(\mathbf{x}-\mathbf{x}_p))_{\iph,j,k} &= \frac{1}{\Delta x }  \, S^{d_x-1}\left(\frac{x_\iph-x_p}{\Delta x}\right)\,S^{d_y}\left(\frac{y_j-y_p}{\Delta y}\right)\,S^{d_z}\left(\frac{z_k-z_p}{\Delta z}\right) \\
        \tcR_2(S_h(\mathbf{x}-\mathbf{x}_p))_{i,\jph,k} &= \frac{1}{\Delta y} \, S^{d_x}\left(\frac{x_i-x_p}{\Delta x}\right)\,S^{d_y-1}\left(\frac{y_\jph-y_p}{\Delta y}\right)\,S^{d_z}\left(\frac{z_k-z_p}{\Delta z}\right) \\
        \tcR_2(S_h(\mathbf{x}-\mathbf{x}_p))_{i,j,\kph} &=\frac{1}{\Delta z}  \, S^{d_x}\left(\frac{x_i-x_p}{\Delta x}\right)\,S^{d_y}\left(\frac{y_j-y_p}{\Delta y}\right)\,S^{d_z-1}\left(\frac{z_\kph-z_p}{\Delta z}\right) \\
        \tcR_3(S_h(\mathbf{x}-\mathbf{x}_p))_{i,j,k} &= S^{d_x}\left(\frac{x_i-x_p}{\Delta x}\right)\,S^{d_y}\left(\frac{y_j-y_p}{\Delta y}\right)\,S^{d_z}\left(\frac{z_k-z_p}{\Delta z}\right).
    \end{aligned}
    $$
    We observe that the terms of the reduction are products of \textbf{node centered splines} of the form $S^{d_x}\left(\frac{x_i-x_p}{\Delta x}\right)$ and of
    \textbf{cell centered splines} of the form $S^{d_x-1}\left(\frac{x_\iph-x_p}{\Delta x}\right)$.

    \subsection{Smoothed particle quantities}

    The particle distribution function is of the form
    $$ f(t,\mathbf{x},\mathbf{v}) = \sum_p w_p S_h(\mathbf{x}-\mathbf{x}_p(t))\delta (\mathbf{v}-\mathbf{v}_p(t)).$$
    It follows that the particle charge and current densities are respectively
    $$\rho(t,\mathbf{x}) = \sum_p q_p w_p S_h(\mathbf{x}-\mathbf{x}_p(t)), ~~~~ \mathbf{J}(t,\mathbf{x}) = \sum_p q_p w_p \mathbf{v}_p(t) S_h(\mathbf{x}-\mathbf{x}_p(t)).$$
    In the GEMPIC theory, the particle charge density $rho$ is a 3-form on the dual mesh and is characterized by its cell integrals on the dual mesh $\tcR_3(\rho)$ and the particle current density is a 2-form on the dual mesh characterized by its fluxes through all the faces of the mesh $\tcR_2( \mathbf{J})$.


    It follows by linearity of the reduction operators that the degrees of freedom (the cell integrals) of $\rho$, which are its coefficients in $\tilde{ \cC}_3$, are 
    $$\tilde{\rho}_{i,j,k} = \sum_p q_p w_p S^{d_x}\left(\frac{x_i-x_p}{\Delta x}\right)\,S^{d_y}\left(\frac{y_j-y_p}{\Delta y}\right)\,S^{d_z}\left(\frac{z_k-z_p}{\Delta z}\right)$$
    and similarly the coefficients of $ \mathbf{J}$ in $\tilde{ \cC}_2$, are 
    $$
    \begin{aligned}
    \tilde{J}_{\iph,j,k} &= \frac{1}{\Delta x}  \, \sum_p q_p w_p v_{p,x}(t) S^{d_x-1}\left(\frac{x_\iph-x_p}{\Delta x}\right)\,S^{d_y}\left(\frac{y_j-y_p}{\Delta y}\right)\,S^{d_z}\left(\frac{z_k-z_p}{\Delta z}\right) \\
    \tilde{J}_{i,\jph,k} &= \frac{1}{\Delta y }  \, \sum_p q_p w_p v_{p,y}(t) S^{d_x}\left(\frac{x_i-x_p}{\Delta x}\right)\,S^{d_y-1}\left(\frac{y_\jph-y_p}{\Delta y}\right)\,S^{d_z}\left(\frac{z_k-z_p}{\Delta z}\right) \\
    \tilde{J}_{i,j,\kph} &= \frac{1}{\Delta z} \, \sum_p q_p w_p v_{p,z}(t) S^{d_x}\left(\frac{x_i-x_p}{\Delta x}\right)\,S^{d_y}\left(\frac{y_j-y_p}{\Delta y}\right)\,S^{d_z-1}\left(\frac{z_\kph-z_p}{\Delta z}\right)
    \end{aligned}
    $$
    which are the components corresponding respectively to $J_x$, $J_y$, $J_z$ 

    \subsection{Smoothed electromagnetic fields}

The discrete electric field $\bE$ is characterized by its integrals on the edges of the primal grid denoted by $\arrE$, where 
\begin{itemize}
    \item $\arrE_{\iph,j,k} \approx \int_{x_i}^{x_{i+1}} E_x(x,y_j,z_k)\dd x $
    \item $\arrE_{i,\jph,k} \approx \int_{y_j}^{y_{j+1}} E_y(x_i,y,z_k)\dd y $
    \item $\arrE_{\iph,j,k} \approx \int_{z_k}^{z_{k+1}} E_z(x_i,y_j,z)\dd z $
\end{itemize}
Based on the Finite Difference Lagrangian from which the electric field is derived, its smoothed version used for evaluating it at the particle positions is based on the duality of the edges of the primal grid
and the faces of the dual grid. Therefore, the smoothing function is projected on the faces of the dual grid with the $\tcR_2$ reduction operator. This yields the following expression of $\bE$ at the particle positions
$$
\begin{aligned}
    E_x(\mathbf{x}_p) &= \frac{1}{\Delta x}  \, \sum_{i,j,k} \arrE_{\iph,j,k} S^{d_x-1}\left(\frac{x_\iph-x_p}{\Delta x}\right)\,S^{d_y}\left(\frac{y_j-y_p}{\Delta y}\right)\,S^{d_z}\left(\frac{z_k-z_p}{\Delta z}\right), \\
    E_y(\mathbf{x}_p) &=\frac{1}{\Delta y} \, \sum_{i,j,k} \arrE_{i,\jph,k} S^{d_x}\left(\frac{x_i-x_p}{\Delta x}\right)\,S^{d_y-1}\left(\frac{y_\jph-y_p}{\Delta y}\right)\,S^{d_z}\left(\frac{z_k-z_p}{\Delta z}\right), \\
    E_z(\mathbf{x}_p) &=\frac{1}{\Delta x} \, \sum_{i,j,k} \arrE_{i,j,\kph} S^{d_x}\left(\frac{x_i-x_p}{\Delta x}\right)\,S^{d_y}\left(\frac{y_j-y_p}{\Delta y}\right)\,S^{d_z-1}\left(\frac{z_\kph-z_p}{\Delta z}\right). 
\end{aligned}
$$

In the same way the discrete magnetic field $\bB$ is characterized by its fluxes through the faces of the primal grid denoted by $\arrE$, where 
\begin{itemize}
    \item $\arrB_{\iph,j,k} \approx \int_{y_j}^{y_{j+1}}\int_{z_k}^{z_{k+1}} B_x(x_i,y,z)\dd y \dd z $
    \item $\arrB_{i,\jph,k} \approx \int_{x_i}^{x_{i+1}}\int_{z_k}^{z_{k+1}} B_y(x,y_j,z)\dd x \dd z $
    \item $\arrB_{\iph,j,k} \approx \int_{x_i}^{x_{i+1}}\int_{y_j}^{y_{j+1}} B_z(x,y,z_k)\dd x\dd y $
\end{itemize}
For the magnetic field, based on the Lagrangian, the smoothing function is projected on the faces of the dual grid with the $\tcR_1$ reduction operator. 
This yields the following expression of $\bB$ at the particle positions
$$
\begin{aligned}
    B_x(\mathbf{x}_p) &= \frac{1}{\Delta y \Delta z} \sum_{i,j,k} \arrB_{i,\jph,\kph} \,S^{d_x}\left(\frac{x_i-x_p}{\Delta x}\right)\,S^{d_y-1}\left(\frac{y_\jph-y_p}{\Delta y}\right)\,S^{d_z-1}\left(\frac{z_\kph-z_p}{\Delta z}\right), \\
    B_y(\mathbf{x}_p) &= \frac{1}{\Delta x \Delta z} \sum_{i,j,k} \arrB_{\iph,j,\kph} \,S^{d_x-1}\left(\frac{x_\iph-x_p}{\Delta x}\right)\,S^{d_y}\left(\frac{y_j-y_p}{\Delta y}\right)\,S^{d_z-1}\left(\frac{z_\kph-z_p}{\Delta z}\right), \\
    B_z(\mathbf{x}_p) &= \frac{1}{\Delta x \Delta y} \sum_{i,j,k} \arrB_{\iph,\jph,k} \,S^{d_x-1}\left(\frac{x_\iph-x_p}{\Delta x}\right)\,S^{d_y-1}\left(\frac{y_\jph-y_p}{\Delta y}\right)\,S^{d_z}\left(\frac{z_k-z_p}{\Delta z}\right). 
\end{aligned}
$$

\subsection{Exact computation of time integrated $\bJ_x$, $v_{p,x}B_y(x_p)$ and $v_{p,x}B_z(x_p)$ }

One step of the hamiltonian splitting algorithm, consists in pushing the particles successively in the 3 axis dimensions and updating positions, velocities and electric field leaving the magnetic field constant.
Let us describe this algorithm when the particles are pushed in the $x$-direction. The two other directions are handled similarly.  

In this step the following system needs to be solved
$$
\begin{aligned}
    \fract{x_p}{t} &= v_{p,x} \\
    \fract{v_{y,p}}{t} &= - v_{p,x}(t)B_z(x_p(t) ,y_p,z_p) \\
    \fract{v_{z,p}}{t} &= v_{p,x}B_y(x_p(t),y_p,z_p) \\
    \fract{\tilde{\arrD}_{\iph,j,k}}{t} &=  -\tilde{\arrJ}_{\iph,j,k} \\ 
    &= -\frac{1}{\Delta x}  \, \sum_p q_p w_p v_{p,x}(t) \, S^{d_x-1}\left(\frac{x_\iph-x_p(t)}{\Delta x}\right)\,S^{d_y}\left(\frac{y_j-y_p}{\Delta y}\right)\,S^{d_z}\left(\frac{z_k-z_p}{\Delta z}\right)
\end{aligned}
$$
while the other variables, not appearing on the left-hand side, stay constant.
Using the first equation $v_{p,x}=\fract{x_p}{t}$, we can write the others as exact time derivatives by introducing the primitive of the spline $S_{\Delta x}^{d_x-1}$,
that we denote by $\check{S}_{\Delta x}^{d_x-1}$. The integration constant is irrelevant as only differences of these primitives will be needed.
Indeed,
$$
\begin{aligned}
    v_{p,x}(t) B_z(x_p(t) ,y_p,z_p) &=   
    \frac{1}{\Delta y} \sum_{i,j,k} \arrB_{\iph,\jph,k} \fract{\check{S}^{d_x-1}}{t}\left(\frac{x_\iph-x_p}{\Delta x}\right)\,S^{d_y-1}\left(\frac{y_\jph-y_p}{\Delta y}\right)\,S^{d_z}\left(\frac{z_k-z_p}{\Delta z}\right) \\
    v_{p,x}(t)  B_y(x_p(t) ,y_p,z_p) &= 
    \frac{1}{\Delta z} \sum_{i,j,k} \arrB_{\iph,j,\kph} \fract{\check{S}^{d_x-1}}{t}\left(\frac{x_\iph-x_p}{\Delta x}\right)\,S^{d_y}\left(\frac{y_j-y_p}{\Delta y}\right)\,S^{d_z-1}\left(\frac{z_\kph-z_p}{\Delta z}\right), \\
    v_{p,x}(t) S_{\Delta x}^{d_x-1}(x_\iph-x_p(t)) &\,S^{d_y}\left(\frac{y_j-y_p}{\Delta y}\right)\,S^{d_z}\left(\frac{z_k-z_p}{\Delta z}\right) = \\ 
    & \Delta x \fract{\check{S}^{d_x-1}}{t}\left(\frac{x_\iph-x_p}{\Delta x}\right)\,S^{d_y}\left(\frac{y_j-y_p}{\Delta y}\right)\,S^{d_z}\left(\frac{z_k-z_p}{\Delta z}\right) .
\end{aligned}
$$
Then, integrating the original system in time between $t_0$ and $t_1$ yields without further approximation
$$
\begin{aligned}
    x_p&(t_1) = x_p(t_0) + (t_1-t_0) v_{p,x} \\
    v_{y,p}&(t_1) = v_{y,p}(t_0) -  
    \frac{1}{\Delta y} \sum_{i,j,k} \arrB_{\iph,\jph,k} \times \\ &\left(\check{S}^{d_x-1}\left(\frac{x_\iph-x_p(t_1)}{\Delta x}\right)-\check{S}^{d_x-1}\left(\frac{x_\iph-x_p(t_0)}{\Delta x}\right)  \right) 
      S^{d_y-1}\left(\frac{y_\jph-y_p}{\Delta y}\right)\,S^{d_z}\left(\frac{z_k-z_p}{\Delta z}\right)\\
    v_{z,p}&(t_1) =  v_{z,p}(t_0) + 
    \frac{1}{\Delta z} \sum_{i,j,k} \arrB_{\iph,j,\kph}  \times \\ &\left(\check{S}^{d_x-1}\left(\frac{x_\iph-x_p(t_1)}{\Delta x}\right)-\check{S}^{d_x-1}\left(\frac{x_\iph-x_p(t_0)}{\Delta x}\right)\right) 
     S^{d_y}\left(\frac{y_j-y_p}{\Delta y}\right)\,S^{d_z-1}\left(\frac{z_\kph-z_p}{\Delta z}\right) \\
    \tilde{\arrD}_{\iph,j,k} & (t_1) =  \tilde{\arrD}_{\iph,j,k}(t_0) - 
    \frac{1}{\Delta x}  \, \sum_p q_p w_p v_{p,x}(t)  \times \\ &\left(\check{S}^{d_x-1}\left(\frac{x_\iph-x_p(t_1)}{\Delta x}\right)-\check{S}^{d_x-1}\left(\frac{x_\iph-x_p(t_0)}{\Delta x}\right)\right)  
    S^{d_y}\left(\frac{y_j-y_p}{\Delta y}\right)\,S^{d_z}\left(\frac{z_k-z_p}{\Delta z}\right)
\end{aligned}
$$
